**attio_client.py — Add lists methods:**

Attio Lists API:
- `GET /v2/lists` — list all lists. Returns `{"data": [...]}`.
- `GET /v2/lists/{list_id}` — get list
- `POST /v2/lists` — create list. Body: `{"data": {"name": "...", "parent_object": "people"}}`.
- `PATCH /v2/lists/{list_id}` — update list. Body: `{"data": {"name": "..."}}`.
- `GET /v2/lists/{list_id}/views` — list views for a list

```python
# ── Lists operations ──────────────────────────────────────────────────
def list_lists(self) -> dict[str, Any]:
    return self._request("GET", "/lists")

def get_list(self, list_id: str) -> dict[str, Any]:
    return self._request("GET", f"/lists/{list_id}")

def create_list(self, name: str, parent_object: str) -> dict[str, Any]:
    return self._request("POST", "/lists", json={"data": {"name": name, "parent_object": parent_object}})

def update_list(self, list_id: str, name: str) -> dict[str, Any]:
    return self._request("PATCH", f"/lists/{list_id}", json={"data": {"name": name}})

def list_list_views(self, list_id: str) -> dict[str, Any]:
    return self._request("GET", f"/lists/{list_id}/views")
```

**attio_client.py — Add entries methods:**

Attio List Entries API:
- `POST /v2/lists/{list_id}/entries/query` — list/query entries (same offset pagination as records). Body: `{"filter": {...}, "sorts": [...], "limit": N, "offset": N}`.
- `GET /v2/lists/{list_id}/entries/{entry_id}` — get entry
- `POST /v2/lists/{list_id}/entries` — create entry. Body: `{"data": {"parent_record_id": "rec_xxx", "values": {...}}}`.
- `PATCH /v2/lists/{list_id}/entries/{entry_id}` — update entry (append multiselect). Body: `{"data": {"values": {...}}}`.
- `PUT /v2/lists/{list_id}/entries/{entry_id}` — overwrite entry (replace multiselect). Same body.
- `DELETE /v2/lists/{list_id}/entries/{entry_id}` — delete entry
- `PUT /v2/lists/{list_id}/entries` — assert entry. Body: `{"data": {"parent_record_id": "rec_xxx", "values": {...}}}`. Upserts by parent record.

CRITICAL: `list_entries` must use `offset_paginator` exactly like `list_records`:
```python
# ── List Entries operations ───────────────────────────────────────────
def list_entries(self, list_id: str, limit: int = 500, all_pages: bool = False, filter: dict[str, Any] | None = None, sorts: list[dict[str, Any]] | None = None) -> Iterator[dict[str, Any]]:
    base_body: dict[str, Any] = {}
    if filter: base_body["filter"] = filter
    if sorts: base_body["sorts"] = sorts
    return offset_paginator(request_fn=self._request, method="POST", path=f"/lists/{list_id}/entries/query", base_body=base_body, limit=limit, all_pages=all_pages)

def get_entry(self, list_id: str, entry_id: str) -> dict[str, Any]:
    return self._request("GET", f"/lists/{list_id}/entries/{entry_id}")

def create_entry(self, list_id: str, parent_record_id: str, values: dict[str, Any] | None = None) -> dict[str, Any]:
    data: dict[str, Any] = {"parent_record_id": parent_record_id}
    if values: data["values"] = values
    return self._request("POST", f"/lists/{list_id}/entries", json={"data": data})

def update_entry(self, list_id: str, entry_id: str, values: dict[str, Any], overwrite: bool = False) -> dict[str, Any]:
    method = "PUT" if overwrite else "PATCH"
    return self._request(method, f"/lists/{list_id}/entries/{entry_id}", json={"data": {"values": values}})

def delete_entry(self, list_id: str, entry_id: str) -> dict[str, Any]:
    return self._request("DELETE", f"/lists/{list_id}/entries/{entry_id}")

def assert_entry(self, list_id: str, parent_record_id: str, values: dict[str, Any] | None = None) -> dict[str, Any]:
    data: dict[str, Any] = {"parent_record_id": parent_record_id}
    if values: data["values"] = values
    return self._request("PUT", f"/lists/{list_id}/entries", json={"data": data})
```
