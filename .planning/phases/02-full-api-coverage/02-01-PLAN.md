**attio_client.py — Add notes methods (after the records section):**

Attio Notes API endpoints:
- `POST /v2/notes` — create note. Body: `{"data": {"parent_object": "people", "parent_record_id": "rec_xxx", "title": "...", "content": "...", "format": "plaintext"}}`. Returns note object.
- `GET /v2/notes/{note_id}` — get single note
- `GET /v2/notes` — list notes. Query params: `parent_object`, `parent_record_id` (optional filters). Returns `{"data": [...]}`.
- `PATCH /v2/notes/{note_id}` — update note title/content. Body: `{"data": {"title": "...", "content": "..."}}`.
- `DELETE /v2/notes/{note_id}` — delete note

Add these methods to AttioClient class:
```python
# ── Notes operations ──────────────────────────────────────────────────
def create_note(self, parent_object: str, parent_record_id: str, title: str, content: str, format: str = "plaintext") -> dict[str, Any]:
    return self._request("POST", "/notes", json={"data": {"parent_object": parent_object, "parent_record_id": parent_record_id, "title": title, "content": content, "format": format}})

def get_note(self, note_id: str) -> dict[str, Any]:
    return self._request("GET", f"/notes/{note_id}")

def list_notes(self, parent_object: str | None = None, parent_record_id: str | None = None) -> dict[str, Any]:
    params = {}
    if parent_object: params["parent_object"] = parent_object
    if parent_record_id: params["parent_record_id"] = parent_record_id
    return self._request("GET", "/notes", params=params)

def update_note(self, note_id: str, title: str | None = None, content: str | None = None) -> dict[str, Any]:
    data: dict[str, Any] = {}
    if title is not None: data["title"] = title
    if content is not None: data["content"] = content
    return self._request("PATCH", f"/notes/{note_id}", json={"data": data})

def delete_note(self, note_id: str) -> dict[str, Any]:
    return self._request("DELETE", f"/notes/{note_id}")
```

**attio_client.py — Add tasks methods:**

Attio Tasks API endpoints:
- `POST /v2/tasks` — create task. Body: `{"data": {"content": "...", "deadline_at": "ISO8601", "is_completed": false, "assignees": [{"referenced_actor_type": "workspace-member", "referenced_actor_id": "..."}], "linked_records": [{"target_object": "people", "target_record_id": "rec_xxx"}]}}`.
- `GET /v2/tasks/{task_id}` — get single task
- `GET /v2/tasks` — list tasks. Query params: `linked_object`, `linked_record_id`, `assignee`, `is_completed`.
- `PATCH /v2/tasks/{task_id}` — update task (content, deadline, is_completed, assignees)
- `DELETE /v2/tasks/{task_id}` — delete task

Add these methods:
```python
# ── Tasks operations ──────────────────────────────────────────────────
def create_task(self, content: str, deadline_at: str | None = None, is_completed: bool = False, assignees: list[dict[str, Any]] | None = None, linked_records: list[dict[str, Any]] | None = None) -> dict[str, Any]:
    data: dict[str, Any] = {"content": content, "is_completed": is_completed}
    if deadline_at: data["deadline_at"] = deadline_at
    if assignees: data["assignees"] = assignees
    if linked_records: data["linked_records"] = linked_records
    return self._request("POST", "/tasks", json={"data": data})

def get_task(self, task_id: str) -> dict[str, Any]:
    return self._request("GET", f"/tasks/{task_id}")

def list_tasks(self, linked_object: str | None = None, linked_record_id: str | None = None, assignee: str | None = None, is_completed: bool | None = None) -> dict[str, Any]:
    params: dict[str, Any] = {}
    if linked_object: params["linked_object"] = linked_object
    if linked_record_id: params["linked_record_id"] = linked_record_id
    if assignee: params["assignee"] = assignee
    if is_completed is not None: params["is_completed"] = str(is_completed).lower()
    return self._request("GET", "/tasks", params=params)

def update_task(self, task_id: str, content: str | None = None, deadline_at: str | None = None, is_completed: bool | None = None, assignees: list[dict[str, Any]] | None = None) -> dict[str, Any]:
    data: dict[str, Any] = {}
    if content is not None: data["content"] = content
    if deadline_at is not None: data["deadline_at"] = deadline_at
    if is_completed is not None: data["is_completed"] = is_completed
    if assignees is not None: data["assignees"] = assignees
    return self._request("PATCH", f"/tasks/{task_id}", json={"data": data})

def delete_task(self, task_id: str) -> dict[str, Any]:
    return self._request("DELETE", f"/tasks/{task_id}")
```
